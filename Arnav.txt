void solve(){
    int n;cin>>n;
    vi a(n);
    read(a);
    int ans = 0;
    for(int i = 1;i<256;i++){
        vi temp(256,0);
        temp[0]++;
        int cur_xor = 0;
        bool ok = 0;
        for(int j = 0;j<n;j++){
            cur_xor ^= a[j];
            temp[cur_xor]++;
            if(temp[cur_xor^i]){
                ok = 1;
                break;
            }
        }
        if(ok) ans = max(ans,i);
    }
    cout<<ans<<endl;
}
